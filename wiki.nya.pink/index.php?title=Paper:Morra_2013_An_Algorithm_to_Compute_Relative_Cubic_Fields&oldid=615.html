This paper describes two notions of reduction for binary cubic forms: Julia reduction and τ {\displaystyle \tau } -reduction. Julia reduction depends on the theory of reduced hermitian forms. If F {\displaystyle F} is Julia-reduced, then its τ {\displaystyle \tau } -reduced form carries additional interesting properties.

// generate all integers within a certain absolute value
pair<IntegerRingElement *, int> get_ints_within (double R) {
  int max_size = 4 * ceil(pow(R, 2));
  // IntegerRingElement ints[max_size];

  IntegerRingElement * ints = (IntegerRingElement*) malloc(sizeof (IntegerRingElement) * max_size);

  int size = 0;

  if (-D % 4 != 1) {
    ints[size++] = IntegerRingElement(0, 0);

    for (int a = 1; a < R; a++) {
      // all nonzero integers on x-axis
      ints[size++] = IntegerRingElement( a, 0);
      ints[size++] = IntegerRingElement(-a, 0);

      for (int b = 1; b < R / sqrt(D); b++) {
        if (pow(a,2) + D * pow(b,2) <= pow(R,2)) {
          ints[size++] = IntegerRingElement( a, b);
          ints[size++] = IntegerRingElement(-a, b);
          ints[size++] = IntegerRingElement( a,-b);
          ints[size++] = IntegerRingElement(-a,-b);
        }
      }
    }

    for (int b = 1; b < R / sqrt(D); b++) {
      // all nonzero integers on y-axis
      ints[size++] = IntegerRingElement(0, b);
      ints[size++] = IntegerRingElement(0, -b);
    }

    return make_pair(ints, size);
  } else {
    return make_pair(ints, size);
  }
}

Code for computing roots of cubic equations with high precision, based on durand-kerner algorithm 
#include <iostream>
#include <string>
#include <tuple>
#include <cln/cln.h>
#include <cln/io.h>
using namespace std;
using namespace cln;

string precision = "_10000";
cl_R eps = ("1.0e-5000" + precision).c_str();

// cl_N a = ("-3.2" + precision).c_str();
// cl_N b = ("-12.6" + precision).c_str();
// cl_N c = ("18.0" + precision).c_str();

cl_N f (cl_N a, cl_N b, cl_N c, cl_N d, cl_N x) {
  return a*expt(x,3) + b*expt(x,2) + c*x + d;
};

tuple<cl_N, cl_N, cl_N> calculate_roots(cl_N a, cl_N b, cl_N c, cl_N d) {

  // complex number that is neither real, nor a root of unity.
  cl_N z = complex(
      ("0.4" + precision).c_str(),
      ("0.9" + precision).c_str()
    );

  // Initial guesses for roots of polynomial f    
  cl_N p = 1;
  cl_N q = z;
  cl_N r = expt(z, 2);

  int i = 0;
  while (
    (abs(f(a,b,c,d,p)) > eps) ||
    (abs(f(a,b,c,d,q)) > eps) ||
    (abs(f(a,b,c,d,r)) > eps)
  ) {
    p = p - f(a,b,c,d,p) / ((p-q) * (p-r));
    q = q - f(a,b,c,d,q) / ((q-p) * (q-r));
    r = r - f(a,b,c,d,r) / ((r-p) * (r-q));
    i = i + 1;
  }

  cout << i << endl;

  cout << "p = "
       << cl_float(realpart(p), float_format(30))
       << " + i * "
       << cl_float(imagpart(p), float_format(15))
       << endl;

  cout << "q = "
       << cl_float(realpart(q), float_format(30))
       << " + i * "
       << cl_float(imagpart(q), float_format(15))
       << endl;

  cout << "r = "
       << cl_float(realpart(r), float_format(30))
       << " + i * "
       << cl_float(imagpart(r), float_format(15))
       << endl;

  return make_tuple(p, q, r);
}
